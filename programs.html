package main

import "fmt"

var p = fmt.Println

func main() {

var reverse [4]string

var test = "test"

var j = len(test)

for i := 0; i < len(test); i++ {

p("Index of: " , i, string(test[i]))

j--

reverse[j] = string(test[i])

}

p()

for j := 0; j < len(reverse); j++ {

p("Index of: ", j, string(reverse[j]))

}

}
