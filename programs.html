Building/Running Programs

Once you have installed the Go compiler following the instructions on the home page, you can run the following programs! In order to run a .go file, in the command line type the following command

$ go run filename.go

This will run the .go program. If you want to build the program into an executable, type the command

$ go build filename.go

This will build the program into a binary file that can be run by typing

$ ./filename

This will run the program. Programs 3 and 4 require user input.

Program 2: Reverse String

Contributor: Brandon Burr

package main

import "fmt"

var p = fmt.Println

func main() {

var reverse [4]string

var test = "test"

var j = len(test)

for i := 0; i < len(test); i++ {

p("Index of: " , i, string(test[i]))

j--

reverse[j] = string(test[i])

}

p()

for j := 0; j < len(reverse); j++ {

p("Index of: ", j, string(reverse[j]))

}

}

Output:

Index of: 0 t

Index of: 1 e

Index of: 2 s

Index of: 3 t

--------------

Index at: 0 t

Index at: 1 s

Index at: 2 e

Index at: 3 t

Program 5: Sum

Contributor: Nicholas Petitmermet

package main

import (

"fmt"

)

func sum() func() int {

sum := 0

iteration := 0

return func() int {

sum = sum + iteration

iteration++

return sum

}

}

func main() {

result := 0

sum := sum()

for i := 0; i <= 5; i++ {

result = sum()

}

fmt.Println(result)

}

15

Program 6: Compute Powers

package main

"fmt"

)

func computePower(x int, y int) int {

sum := 1

for i := 0; i < y; i++ {

sum *= x

}

return sum

}

func main() {

}

Program 7: Recursively Compute Powers

Contributor: Josh Berkenmeier

package main

import (

"fmt"

)

func recursivePower(x int, y int) int {

if y < 1 {

return 1

} else {

return x * recursivePower(x, y-1)

}

}

func main() {

fmt.Println(recursivePower(2, 0))

fmt.Println(recursivePower(2, 6))

}

Output:

1

64

program 8: Decimal to Hexadecimal Converter

Contributor: Brandon Burr

package main

import (

"fmt"

)

var p = fmt.Println

var print = fmt.Print

func hex(n int) {

var temp, number, count int

number = n

var reverseHex [8]int

count = 0

for i:=0; number>0; i++ {

temp = number % 16

reverseHex[i] = temp

number = number/16

count++

}

hexarray := make([]int, count)

count = 0

print("Hexidecimal Equivalent: 0x")

for j:= len(hexarray)-1; j>= 0; j-- {

hexarray[count] = reverseHex[j]

if(hexarray[count] == 10){

print("a")

}else if(hexarray[count] == 11){

print("b")

}else if(hexarray[count] == 12){

print("c")

}else if(hexarray[count] == 13){

print("d")

}else if(hexarray[count] == 14){

print("e")

}else if(hexarray[count] == 15){

print("f")

}else{

print(hexarray[count])

}

}

p()

}

func main() {

for{

var i int

print("Enter an integer value: "

fmt.Scanln(&i)

hex(i)

}

}
