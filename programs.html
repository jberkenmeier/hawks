Program 2: Reverse String

package main

import "fmt"

var p = fmt.Println

func main() {

var reverse [4]string

var test = "test"

var j = len(test)

for i := 0; i < len(test); i++ {

p("Index of: " , i, string(test[i]))

j--

reverse[j] = string(test[i])

}

p()

for j := 0; j < len(reverse); j++ {

p("Index of: ", j, string(reverse[j]))

}

}

Output:

Index of: 0 t

Index of: 1 e

Index of: 2 s

Index of: 3 t

--------------

Index at: 0 t

Index at: 1 s

Index at: 2 e

Index at: 3 t

Program 4: Writitng to a file

package main

import (

"bufio"

"fmt"

"os"

"strings"

)

func main() {

f, _ := os.Create("output.txt")

read := bufio.NewReader(os.Stdin)

fmt.Println("Type what you would like to go in output.txt")

text, _ := read.ReadString('\n')

text = strings.Replace(text, "\n" , "", -1)

filetext := [] byte (text)

f.Write(filetext)

}

Program 5: Sum

package main

import (

"fmt"

)

func sum() func() int {

sum := 0

iteration := 0

return func() int {

sum = sum + iteration

iteration++

return sum

}

}

func main() {

result := 0

sum := sum()

for i := 0; i <= 5; i++ {

result = sum()

}

fmt.Println(result)

}

15
